1.class

class可以當作是模組，產出具有相似特性的物件

class Animal():
    def __init__(self, name):
        self.name = name
x = Animal("dog")

print(x.name)

dog

class Account:

    def __init__(self,name,number):
        self.name = name
        self.number = number
        self.balance = 0

    def deposit(self,amount):
        if amount<0:
            raise '輸入錯誤'
        self.balance += amount

    def withdraw(self,amount):
        if amount<=self.balance:
            self.balance -= amount
        else:
            raise RuntimeError('餘額不足')

account = Account('Andy','123–456–789') 
account.deposit(100)
account.withdraw(30)
account.balance

70

2.decorator
# syntactic sugar 快寫法

class Animal:
    def __init__(self, barking):
        self.barking = barking

    def bark(self):
        print("Bark~")

@Animal
def dog():
    print("woof~")

@Animal
def cat():
    print("meow~")

x = dog
x.barking()

woof~

4.yield
# 常搭配def/for使用，為節省記憶體

def print_range(n):
    i = 0
    while i<n:
        yield i
        i+=1

for i in print_range(6):
    print(i)

0 1 2 3 4 5
